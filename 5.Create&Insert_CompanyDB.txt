-- Create the database
CREATE DATABASE CompanyDatabase;
USE CompanyDatabase;

-- Create tables
CREATE TABLE EMPLOYEE (
SSN INT PRIMARY KEY,
Name VARCHAR(255),
Address VARCHAR(255),
Sex VARCHAR(1),
Salary DECIMAL(10, 2),
SuperSSN INT,
DNo INT
);

CREATE TABLE DEPARTMENT (
DNo INT PRIMARY KEY,
DName VARCHAR(255),
MgrSSN INT,
MgrStartDate DATE
);

CREATE TABLE DLOCATION (
DNo INT PRIMARY KEY,
DLoc VARCHAR(255)
);

CREATE TABLE PROJECT (
PNo INT PRIMARY KEY,
PName VARCHAR(255),
PLocation VARCHAR(255),
DNo INT
);

CREATE TABLE WORKS_ON (
SSN INT,
PNo INT,
Hours INT,
PRIMARY KEY (SSN, PNo),
FOREIGN KEY (SSN) REFERENCES EMPLOYEE(SSN),
FOREIGN KEY (PNo) REFERENCES PROJECT(PNo)
);
