-- Create the database
CREATE DATABASE CollegeDatabase;
USE CollegeDatabase;

-- Create tables
CREATE TABLE STUDENT (
USN VARCHAR(20) PRIMARY KEY,
SName VARCHAR(255),
Address VARCHAR(255),
Phone VARCHAR(20),
Gender VARCHAR(10)
);

CREATE TABLE SEMSEC (
SSID INT PRIMARY KEY,
Sem INT,
Sec VARCHAR(1)
);

CREATE TABLE CLASS (
USN VARCHAR(20),
SSID INT,
PRIMARY KEY (USN, SSID),
FOREIGN KEY (USN) REFERENCES STUDENT(USN),
FOREIGN KEY (SSID) REFERENCES SEMSEC(SSID)
);

CREATE TABLE COURSE (
Subcode VARCHAR(20) PRIMARY KEY,
Title VARCHAR(255),
Sem INT,
Credits INT
);

CREATE TABLE IAMARKS (
USN VARCHAR(20),
Subcode VARCHAR(20),
SSID INT,
Test1 INT,
Test2 INT,
Test3 INT,
FinalIA INT,
PRIMARY KEY (USN, Subcode, SSID),
FOREIGN KEY (USN) REFERENCES STUDENT(USN),
FOREIGN KEY (Subcode) REFERENCES COURSE(Subcode),
FOREIGN KEY (SSID) REFERENCES SEMSEC(SSID)
);
